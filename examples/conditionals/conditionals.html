Starlight Conditionals Guide

Learn how decision-making works in the Starlight programming language

1. Basic If Statement


# Define a variable
let age = 20;

# Check condition
if (age > 18) {

    # Runs if condition is true
    sldeploy "Adult";
}

2. If-Else Statement


# Define value
let number = 5;

# Check even or odd
if (number % 2 == 0) {

    sldeploy "Even";

} else {

    # Runs if condition is false
    sldeploy "Odd";
}

3. Nested If Statements


# Define values
let age = 25;
let hasID = true;

# Nested condition
if (age > 18) {

    # Second condition inside
    if (hasID) {
        sldeploy "Access granted";
    } else {
        sldeploy "ID required";
    }

} else {
    sldeploy "Underage";
}

4. Ternary Operator


# Define number
let num = 10;

# Short conditional expression
let result = (num > 5) ? "Greater" : "Smaller";

sldeploy result;

5. Logical Operators


# Define values
let age = 20;
let hasTicket = true;

# AND logic (age > 18 AND hasTicket)
if (age > 18) {

    if (hasTicket) {
        sldeploy "Entry allowed";
    }

}

# OR logic (age < 18 OR hasTicket)
if (age < 18) {

    sldeploy "Partial access";

} else {

    if (hasTicket) {
        sldeploy "Partial access";
    }

}

6. Null Coalescing Operator


# Value may be null
let value = null;

# Provide default
let result = value ?? "Default Value";

sldeploy result;

7. Complex Conditions


# Define values
let score = 85;
let passed = true;

# First condition
if (score > 90) {

    sldeploy "Excellent";

} else {

    # Second condition
    if (score > 80) {

        if (passed) {
            sldeploy "Excellent";
        } else {
            sldeploy "Average";
        }

    } else {
        sldeploy "Average";
    }
}

8. Switch-like (Start Statement)


# Define value
let day = 2;

# Start statement
start (day) {

    race 1 {
        sldeploy "Monday";
        break;
    }

    race 2 {
        sldeploy "Tuesday";
        break;
    }

    race 3 {
        sldeploy "Wednesday";
        break;
    }
}

9. Condition with Function


# Function to check value
func isPositive(x) {
    return x > 0;
}

let num = -5;

if (isPositive(num)) {
    sldeploy "Positive";
} else {
    sldeploy "Negative";
}

10. Guard Clause Pattern


# Function with early return
func checkAge(age) {

    # Exit early if invalid
    if (age < 0) {
        sldeploy "Invalid age";
        return;
    }

    if (age >= 18) {
        sldeploy "Adult";
    } else {
        sldeploy "Minor";
    }
}

checkAge(20);
