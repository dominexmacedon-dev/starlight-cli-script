Starlight Looping Guide

Learn how loops work in the Starlight programming language

1. While Loop


# Initialize counter
define i = 0;

# Loop runs while condition is true
while (i < 5) {

    # Print current value
    sldeploy "i: " + i;

    # Increase counter
    i = i + 1;
}

2. For Loop


# Loop with initialization, condition, increment
for (let i = 0; i < 5; i = i + 1) {

    # Print current iteration
    sldeploy "i: " + i;
}

3. For-In Loop (Array)


# Create an array
define arr = [10, 20, 30];

# Iterate over each element
for item in arr {

    # Print each value
    sldeploy "Value: " + item;
}

4. Break Statement


# Loop with break condition
for (let i = 0; i < 10; i = i + 1) {

    # Stop loop when i equals 5
    if (i == 5) {
        break;
    }

    # Print values before break
    sldeploy "i: " + i;
}

5. Continue Statement


# Loop with continue
for (let i = 0; i < 5; i = i + 1) {

    # Skip when i equals 2
    if (i == 2) {
        continue;
    }

    # Print remaining values
    sldeploy "i: " + i;
}

6. Nested Loops


# Outer loop runs 3 times
for (let i = 0; i < 3; i = i + 1) {

    # Inner loop runs 2 times for each outer iteration
    for (let j = 0; j < 2; j = j + 1) {

        # Print both counters
        sldeploy "i: " + i + ", j: " + j;
    }
}

7. For-In Loop (Object)


# Create an object
let user = {
    "name": "Alice",
    "age": 25
};

# Iterate over keys
for key in user {

    # Access value using key
    sldeploy key + ": " + user[key];
}

8. Infinite Loop with Break


# Infinite loop
while (true) {

    # Generate random number
    let n = random(1, 10);

    sldeploy "Number: " + n;

    # Exit condition
    if (n == 5) {
        break;
    }
}

9. Reverse Loop


# Loop backwards
for (let i = 5; i > 0; i = i - 1) {

    # Countdown output
    sldeploy "i: " + i;
}

10. Conditional Loop (Even Numbers)


# Loop numbers 0 to 10
for (let i = 0; i <= 10; i = i + 1) {

    # Print only even numbers
    if (i % 2 == 0) {
        sldeploy "Even: " + i;
    }
}
